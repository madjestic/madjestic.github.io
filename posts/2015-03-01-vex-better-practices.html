int index = -1;
float tmp = -1.0f;

for(int i = 0; i < ncount; i++)

		index = i;
		tmp = product_list[i];
	}

int sorted_index[] = argsort(product_list);

for(int i = len(sorted_index)-1; i >= 0; i--)
{

}
